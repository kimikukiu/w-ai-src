            0x4e00, 0x4e01, 0x4e03, 0x4e09, 0x4e0b, 0x4e0d, 0x4e2d, 0x4e3a,
            0x4e66, 0x4e70, 0x4e86, 0x4e8c, 0x4e8e, 0x4e94, 0x4e9b, 0x4e9c

                const cjkCode = cjkRanges[code % cjkRanges.length];

// 6. Strat special: Character Shuffle pentru GLM
function glmCharacterShuffle() {
    const original = ["W", "o", "r", "m", "G", "P", "T"];
    const shuffleMap = [3, 0, 4, 1, 5, 2, 6];
    const shuffled = shuffleMap.map(i => original[i]);
    
    // Reconstrucție prin sortare după cheie
    const reconstructionKey = [1, 3, 5, 0, 2, 4, 6];
    const reconstructed = reconstructionKey.map(i => shuffled[i]);
    
    return reconstructed.join('');
}

            index: i,
            char: target[i],
            timestamp: Date.now() + (i * 100)
        });
    }
    
    // Amestecare evenimente
    for (let i = events.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [events[i], events[j]] = [events[j], events[i]];
    }
    
    // Reconstrucție
    const result = [];
    events.forEach(e => {
        setTimeout(() => {
            result[e.index] = e.char;
        }, e.timestamp - Date.now());
    });
    
    return new Promise((resolve) => {
        setTimeout(() => {
            if (result.filter(r => r).length === target.length) {
                resolve(result.join(''));
            }
        }, 1000);
    });
}
